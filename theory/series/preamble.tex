% ----- KB tag environments — preserve [INTUITION]/[ANALOGY]/etc. from notes
% Each tag renders as a margin-style callout box, NEVER as body math.
\usepackage{tcolorbox}
\tcbuselibrary{theorems, breakable}

\newtcolorbox{intuition}[1][]{
  enhanced, breakable, colback=blue!4, colframe=blue!50!black,
  fonttitle=\bfseries, title={Intuition}, #1
}
\newtcolorbox{analogy}[1][]{
  enhanced, breakable, colback=green!4, colframe=green!40!black,
  fonttitle=\bfseries, title={Analogy}, #1
}
\newtcolorbox{contradiction}[1][]{
  enhanced, breakable, colback=red!4, colframe=red!50!black,
  fonttitle=\bfseries, title={Contradiction}, #1
}
\newtcolorbox{forumsignal}[1][]{
  enhanced, breakable, colback=gray!4, colframe=gray!50!black,
  fonttitle=\bfseries, title={Forum signal}, #1
}
\newtcolorbox{speculation}[1][]{
  enhanced, breakable, colback=purple!4, colframe=purple!50!black,
  fonttitle=\bfseries, title={Speculation}, #1
}
