% xr-hyper enables cross-paper \ref into sibling papers' .aux files.
% Per-paper main.tex declares which siblings to import via \externaldocument
% with a [PN-] prefix on each. Must be loaded AFTER hyperref (xr-hyper's
% package-load-time warning is misleading — the AFTER order is what works
% in this TeXLive). \cref support requires cleveref (loaded below).
\usepackage{xr-hyper}

% Patch \externaldocument: when xr-hyper imports labels from sibling .aux
% files, it expands any macros inside captured label content (theorem
% titles, captions). If a label captured a \glsterm{...}{surface}, the
% expansion runs \pdftooltip{\hyperlink{...}{\textcolor{...}{\dotuline{...}}}}{def},
% and \hyperlink reaches \hyper@@link which xr-hyper has redefined to a
% no-arg form — fatal "Use of \hyper@link@ doesn't match its definition."
% Inside the import group, stub every decorator macro down to its
% visible-text component so the imported label resolves cleanly.
% Note: \dotuline is loaded by ulem (below); we declare the stub here
% before ulem loads since \let to an undefined macro is fine — it just
% silently no-ops until ulem's \dotuline becomes the real binding.
\makeatletter
\let\xr@externaldocument@orig\externaldocument
\renewcommand{\externaldocument}[2][]{%
  \begingroup
  \let\hyperlink\@secondoftwo
  \let\pdftooltip\@firstoftwo
  \let\textcolor\@secondoftwo
  \let\dotuline\@firstofone
  \xr@externaldocument@orig[#1]{#2}%
  \endgroup
}
\makeatother

% ----- glossary linking (clickable cross-link + hover tooltip)
\usepackage{pdfcomment}  % requires TeXLive 2018+; project uses TL2022+
\usepackage[normalem]{ulem}  % \dotuline for glossary-term underline; normalem keeps \emph italic

% \glsterm{key}{surface} renders `surface` in a muted rust tint with a
% dotted underline (easier to scan for hover-able terms in dense prose),
% attaches a hyperlink to the per-paper glossary anchor `glossary:<key>`
% (clickable in every PDF viewer), and a PDF tooltip whose text is the
% term's short definition (rendered on hover in Acrobat / Foxit / evince;
% silent in browser-embedded viewers — degrades gracefully). Tint + dots
% together signal "hoverable" without competing with citation blue.
%
% The short-definition macros \@gls@def@<key>{...} are defined in
% paper-N/glossary-tooltips.tex (auto-generated by mark_glossary_terms.py)
% and \input'd at the top of each paper's main.tex.
\definecolor{glscolor}{HTML}{8B5A2B}
\makeatletter
\newcommand{\glsterm}[2]{%
  \pdftooltip{\hyperlink{glossary:#1}{\textcolor{glscolor}{\dotuline{#2}}}}{\@nameuse{@gls@def@#1}}%
}
% \glsdef{key}{surface} renders IDENTICALLY to \glsterm but semantically
% marks the site where the term is FORMALLY introduced. The glossary
% back-arrow → in the glossary section will jump to the surrounding
% \label{sec:...} of the \glsdef site (preferred over the first
% \glsterm occurrence, which is often a casual §1 name-drop). Use one
% \glsdef per term, at the formalization point.
\newcommand{\glsdef}[2]{\glsterm{#1}{#2}}
\makeatother

% \nogls{...} is an author escape hatch: a no-op renderer whose sole

% to suppress an auto-wrap false positive.

% ----- TODO and margin notes (used during writing pass)
% todonotes drives \deepencite{...} — author-only "this needs deepening"
% annotations. They render as inline orange boxes which DON'T fit in
% tabularx cells (Table 6 in paper-4 §13 was a documented offender: the
% block-element \todo[inline] bled off the page and pushed neighboring
% row content into white space).
%
% Default state: `disable` — \todo / \deepencite are no-ops in the
% published render. The annotations remain in source files (grep
% `\deepencite` to find them) so the author can still pick them up
% during a writing pass; just remove `disable` below to re-enable
% rendering for an authoring/review build.
\usepackage[textsize=footnotesize, color=yellow!40, disable]{todonotes}
\newcommand{\deepencite}[1]{\todo[inline,color=orange!30]{deepen-cite: #1}}
